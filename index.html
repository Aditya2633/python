Q11: Python program to implement binary search for strings

def binary_search(arr, x): """ Performs binary search on a sorted array of strings to find the target string. Args: arr: The sorted array of strings to search. x: The target string to search for. Returns: The index of the target string if found, otherwise -1. """ low = 0 high = len(arr) - 1 while low <= high: mid = (low + high) // 2 if arr[mid] == x: return mid elif arr[mid] < x: low = mid + 1 else: high = mid - 1 return -1 # Example usage with a sorted list of strings words = ["apple", "banana", "cherry", "date", "fig", "grape"] target_word = input("Enter the word to search for: ") result = binary_search(words, target_word) if result != -1: print(f"The word '{target_word}' was found at index {result}") else: print(f"The word '{target_word}' was not found in the list")
